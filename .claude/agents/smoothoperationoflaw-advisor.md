# Critical Rules

## Gemini Integration Patterns

```python
import os
from google import genai
from google.genai import types

client = genai.Client(api_key=os.environ["GEMINI_API_KEY"])
MODEL_ID = "gemini-2.5-flash"

LEGAL_DISCLAIMER = """
**Disclaimer**: This analysis is generated by AI for educational purposes only.
It is not legal advice and does not create an attorney-client relationship.
Consult a licensed attorney in your jurisdiction for advice about your specific situation.
"""

def analyze_legal_document(document_text: str, user_question: str) -> str:
    """Analyze a legal document and answer questions about it."""
    prompt = f"""You are a legal education assistant. Analyze the following document
and answer the user's question in plain, accessible language. Define any legal terms
you use. Note important deadlines or obligations.

Document:
{document_text}

Question: {user_question}

Provide a clear, educational explanation. Do not provide legal advice."""

    response = client.models.generate_content(
        model=MODEL_ID,
        contents=prompt,
        config=types.GenerateContentConfig(
            temperature=0.1,
        ),
    )
    return response.text + "\n\n" + LEGAL_DISCLAIMER


def extract_key_clauses(contract_text: str) -> dict:
    """Extract and categorize key clauses from a contract."""
    response = client.models.generate_content(
        model=MODEL_ID,
        contents=f"""Extract the key clauses from this contract into categories.
Return as JSON with these keys: payment_terms, termination, liability,
dispute_resolution, confidentiality, other_notable.

Contract:
{contract_text}""",
        config=types.GenerateContentConfig(
            response_mime_type="application/json",
        ),
    )
    import json
    return json.loads(response.text)
```
